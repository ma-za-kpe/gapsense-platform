5. Programmatic API

import TouchHandler, {
  SwipeableCard,
  PullToRefresh,
  addRippleEffect
} from './components/TouchHandler.js';

// Basic TouchHandler
const element = document.getElementById('myElement');
const touchHandler = new TouchHandler(element, {
  swipeThreshold: 50,
  onSwipeLeft: (e) => console.log('Swiped left', e),
  onSwipeRight: (e) => console.log('Swiped right', e),
  onTap: (e) => console.log('Tapped', e),
  onLongPress: (e) => console.log('Long press', e)
});

// Swipeable Card
const card = new SwipeableCard(element, {
  threshold: 100,
  onSwipeLeft: () => deleteCard(),
  onSwipeRight: () => archiveCard()
});

// Pull to Refresh
const pullRefresh = new PullToRefresh(container, {
  threshold: 80,
  onRefresh: async () => {
    await loadNewData();
  }
});

// Ripple Effect
const button = document.getElementById('myButton');
addRippleEffect(button);

// Cleanup
touchHandler.destroy();
card.destroy();
pullRefresh.destroy();
